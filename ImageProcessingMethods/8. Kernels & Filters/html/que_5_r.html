a=zeros(100,1);
a(45:55)=1;
% subplot(4,2,1)
% plot(a),title('rect function')
b=fft(a);
b=fftshift(b);
subplot(4,2,1)
plot(abs(b)),xlabel('freq'),title('sinc')

c=ifft(ifftshift(b));
subplot(4,2,2)
plot(real(c)),title('reconstructed rect')
